\section{打印系统相关的信息}
\begin{itemize}
\item 安装：
\begin{lstlisting}
# ubuntu 
$ sudo apt-get install -y screenfetch 

# opensuse 
$ sudo zypper in screenfetch
\end{lstlisting}

\item 运行：
\begin{lstlisting}
$ sudo screenfetch
\end{lstlisting}
