\section{系统配置}
\subsection{更新源}

\begin{itemize}
\item 修改更新源
\begin{lstlisting}
// 以阿里源为例
  
$ sudo cp /etc/apt/sources.list /etc/apt/sources.list.bak  
$ sudo deepin-editor /etc/apt/sources.list

* 将其中的 http://packages.deepin.com/deepin 替换成 http://mirrors.aliyun.com/deepin 
\end{lstlisting}

\item 更新
\begin{lstlisting}
# 更新软件列表
$ sudo apt update  

# 升级软件 
$ sudo apt upgrade 
\end{lstlisting}
\end{itemize}
